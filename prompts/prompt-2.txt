## Dodatkowe funkcje (opcjonalne)

### 1. Tooltips
```python
import tkinter.ttk as ttk

def create_tooltip(widget, text):
    """Dodaj tooltip do widgetu"""
    tooltip = tk.Toplevel()
    tooltip.withdraw()
    tooltip.overrideredirect(True)
    label = tk.Label(tooltip, text=text, background="#FFFFE0", 
                     relief='solid', borderwidth=1, font=('Arial', 9))
    label.pack()
    
    def show_tooltip(event):
        tooltip.deiconify()
        tooltip.geometry(f"+{event.x_root+10}+{event.y_root+10}")
    
    def hide_tooltip(event):
        tooltip.withdraw()
    
    widget.bind('<Enter>', show_tooltip)
    widget.bind('<Leave>', hide_tooltip)
```

### 2. Animacja paska postępu
```python
def _animate_progress_bar(self, target_value):
    """Animacja wypełniania paska postępu"""
    current = 0
    step = target_value / 20  # 20 kroków animacji
    
    def animate():
        nonlocal current
        if current < target_value:
            current += step
            self._update_progress_bar(min(current, target_value))
            self.root.after(30, animate)  # 30ms opóźnienia
    
    animate()
```
